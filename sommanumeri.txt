void sommanumeri(char *stringa){
	int i;
	int j=0;
	for(i=0; stringa[i] != '\0'; i++){
		if(stringa[i]>='0'&&stringa[i]<='9'){
			stringa[j]=stringa[i];
			j++;
		}
		else{
			if(j!=0 &&(stringa[i+1]>='0'&&stringa[i+1]<='9'))
				stringa[j]=' ';
				j++;
		}
	}
		stringa[j] = '\0';
}
