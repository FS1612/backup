 int moduloDiverso(int sequenza[],int l);

		 scanf("%d",&interi[i]);

	 if (moduloDiverso(interi, lunghezza)==1)

 int moduloDiverso(int sequenza[], int l){
	 int j; //*variabile contatore
	 int n; //*variabile contatore
	 int modulo;
	 modulo=1;
	 j=0;
	 for(int i=0; i<l; i++){
		 if (sequenza[i]<0)
			 sequenza[i]=-sequenza[i];
	 }
	 while( j<l-1&&modulo){
		 n=j+1;
		 { while (n<l&&modulo)
			 if(sequenza[j]==sequenza[n])
				 modulo=0;

				 else 
		 n++;}
		 j++;
	 }
	 return modulo;
 }
