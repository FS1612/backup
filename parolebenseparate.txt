void parolebenseparate(char*stringa){
int i;
int j=0;
	for(i=0;stringa[i]!='\0';i++){
		if(stringa[i]>='a'&& stringa[i]<='z'){
			stringa[j]=stringa[i];
			j++;
		}
		else {
			if(j>=0&&(stringa[i+1]>='a'&&stringa[i+1]<='z')){
				stringa[j]='-';
				j++;
		}
	}
}
stringa[j]='\0';
}
